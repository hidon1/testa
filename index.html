    <h1>המוצרים שלנו</h1>
    <div id="error"></div>
    <div id="loading">טוען מוצרים...</div>
    <div id="products"></div>

    <script>
        const API_URL = 'YOUR_API_URL'; // החלף ב-URL האמיתי שלך
        
        async function loadProducts() {
            const loading = document.getElementById('loading');
            const errorEl = document.getElementById('error');
            const productsEl = document.getElementById('products');
            
            try {
                const res = await fetch(API_URL);
                if (!res.ok) throw new Error('שגיאה בטעינת הנתונים');
                
                const products = await res.json();
                
                loading.style.display = 'none';
                productsEl.innerHTML = '';
                
                products.forEach(product => {
                    const card = document.createElement('div');
                    card.className = 'card';
                    card.innerHTML = `
                        <img src="${product.image}" alt="${product.name}" onerror="this.src='https://via.placeholder.com/300x200?text=אין+תמונה'">
                        <div class="card-content">
                            <h3>${product.name}</h3>
                            <div class="price">₪${product.price}</div>
                        </div>
                    `;
                    productsEl.appendChild(card);
                });
            } catch (err) {
                loading.style.display = 'none';
                errorEl.textContent = 'שגיאה: ' + err.message;
                errorEl.style.display = 'block';
            }
        }
        
        // טעינה אוטומטית
        window.onload = loadProducts;
    </script>
</body>
</html>
