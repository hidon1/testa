        function checkout() {

                const price = item.saleActive && item.salePrice ? item.salePrice : item.price;
                message += `- ${item.name} × ${item.quantity || 1} = ${price * (item.quantity || 1)} ₪\n`;
            });
            
            const total = cart.reduce((sum, item) => {
                const price = item.saleActive && item.salePrice ? item.salePrice : item.price;
                return sum + price * (item.quantity || 1);
            }, 0);
            
            message += `\nסה"כ: ${total} ₪`;
